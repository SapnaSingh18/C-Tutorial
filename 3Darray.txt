                                                                              3D Array
                                                                        ********************

                                       In C++, a 3D array is an array of arrays of arrays, where each element is itself
                                   a 2D array. It can be thought of as a cuboid with multiple layers, rows, and columns. Here's
                                          a full example of how to declare, initialize, and use a 3D array in C++:

                                                                       
                                                                        Source Code of 3D Array 
                                                                     ```````````````````````````
                                                                         #include <iostream>
                                                                             int main() 
                                                                            {
                                                  // Declare a 3D array with fixed size (2 layers, 3 rows, and 4 columns)

                                                                               const int layers = 2;
                                                                               const int rows = 3;
                                                                               const int columns = 4;
                                                                      int threeDArray[layers][rows][columns];

                                                             // Initialize the 3D array with some values
  
                                                             for (int layer = 0; layer < layers; layer++)
                                                                            {
                                                                    for (int row = 0; row < rows; row++)
                                                                              {
                                                                for (int col = 0; col < columns; col++) 
                                                                                {
                                              threeDArray[layer][row][col] = layer * rows * columns + row * columns + col;
                                                                                  }
                                                                                }
                                                                            }

                                                               // Display the contents of the 3D array

                                                             std::cout << "3D Array Contents:" << std::endl;
                                                               for (int layer = 0; layer < layers; layer++) 
                                                                                {
                                                            std::cout << "Layer " << layer << ":" << std::endl;
                                                                  for (int row = 0; row < rows; row++)
                                                                               {
                                                                 for (int col = 0; col < columns; col++)
                                                                        {
                                                            std::cout << threeDArray[layer][row][col] << "\t";
                                                                               }
                                                                         std::cout << std::endl;
                                                                                  }
                                                                          std::cout << std::endl;
                                                                              }
                                                                            return 0;
                                                                                 }
